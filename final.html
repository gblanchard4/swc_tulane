library("ggplot2")
gap.in <- read.table("output//combined_gapMinder.tsv", header = TRUE, sep = "\t")

Subset the countries

  • Australia
aus <- subset(gap.in, country == "Australia")
  • New Zealand
nz <- subset(gap.in, country == "New Zealand")
  • Ireland
ire <- subset(gap.in, country == "Ireland")
